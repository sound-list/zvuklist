```javascript
// Откройте F12 → Console и используйте:

player.prevTrack();         // Предыдущий

player.setVolume(0.8);      // Громкость 80%
player.selectTrack(1);      // Выбрать трек 1
player.renderRecommendations(); // Рекомендации
player.renderHistory();     // История
```
